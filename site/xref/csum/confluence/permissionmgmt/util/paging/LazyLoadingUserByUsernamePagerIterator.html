41      private UserAccessor userAccessor;
42      private Iterator usernamePagerIterator;
43  
44  
45      public boolean hasNext() {
46          usernamePagerIterator = getUsernamePagerIterator();

52      }

55          usernamePagerIterator = getUsernamePagerIterator();

60  
61          return null;
62      }
63

66          if (usernamePagerIterator != null) {
67              usernamePagerIterator.remove();
68          }
69      }
70

72          return userAccessor;
73      }
74

76          this.userAccessor = userAccessor;
77      }
78  
79      public Iterator getUsernamePagerIterator() {
80          return usernamePagerIterator;
81      }

84          this.usernamePagerIterator = usernamePagerIterator;
