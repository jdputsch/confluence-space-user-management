53      public void addGroups(List groupNames, ServiceContext context) throws AddException {
54          log.debug("addGroups() called. groupName='" + StringUtil.convertCollectionToCommaDelimitedString(groupNames) + "'");
55          Space space = context.getSpace();
56  
57          List success = new ArrayList();
58          List alreadyExisted = new ArrayList();
59  
60          for (int i = 0; i < groupNames.size(); i++) {
61              String groupName = (String) groupNames.get(i);
62              if (userAccessor.getGroup(groupName) == null) {
63  
64                  Group vGroup = userAccessor.addGroup(groupName);
65                  log.debug("created " + groupName);
66                  success.add(groupName);
67  
68                  //If group exists then set all required permissions
69                  if (vGroup != null) {
70                      SpacePermission perm = new SpacePermission(SpacePermission.VIEWSPACE_PERMISSION, space, vGroup.getName());
71                      space.addPermission(perm);
72                      log.debug("added viewspace perm to " + groupName);
73                  }
74              } else {
75                  alreadyExisted.add(groupName);
76              }
77          }
78  
79          if (alreadyExisted.size() > 0) {
80              String msg = "";
81              String concat = "";
82              if (alreadyExisted.size() > 0) {
83                  msg += context.getText("manager.error.groupsalreadyexisted") + ": " +
84                          StringUtil.convertCollectionToCommaDelimitedString(alreadyExisted) + ".";
85                  concat = " ";
86              }
87  
88              if (success.size() > 0) {
89                  msg += concat;
90                  msg += context.getText("manager.error.addgroupspartialsuccess") + ": " +
91                          StringUtil.convertCollectionToCommaDelimitedString(success) + ".";
92              }
93  
94              throw new AddException(msg);
95          }
96      }

98      public void removeGroups(List groupNames, ServiceContext context) throws RemoveException {
99          log.debug("removeGroup() called. groupNames are " + StringUtil.convertCollectionToCommaDelimitedString(groupNames));
100         List didNotExist = new ArrayList();
101         List badGroupNames = new ArrayList();
102         List success = new ArrayList();
103 
104         CustomPermissionConfiguration config = getCustomPermissionConfiguration();
105         String spaceKey = context.getSpace().getKey();
106         String prefix = GroupNameUtil.replaceSpaceKey(config.getNewGroupNameCreationPrefixPattern(), spaceKey);
107         String suffix = GroupNameUtil.replaceSpaceKey(config.getNewGroupNameCreationSuffixPattern(), spaceKey);
108 
109         //Remove Selected Groups
110         for (Iterator iterator = groupNames.iterator(); iterator.hasNext();) {
111             String grpName = (String) iterator.next();
112             boolean isPatternMatch = GroupNameUtil.doesGroupMatchPattern(grpName, prefix, suffix);
113 
114             // Space admin should not be able to delete any groups whose names begin with "confluence"
115             if (!grpName.startsWith("confluence") && isPatternMatch) {
116                 Group group = userAccessor.getGroup(grpName);
117                 if (group != null) {
118                     userAccessor.removeGroup(group);
119                     success.add(grpName);
120                 } else {
121                     didNotExist.add(grpName);
122                 }
123             } else {
124                 log.debug("Not deleting group '" + grpName + "', as either it started with 'confluence', didn't start with '" + prefix + "', or didn't end with '" + suffix + "'");
125                 badGroupNames.add(grpName);
126             }
127         }
128 
129         // if we failed, throw exception
130         if (badGroupNames.size() > 0 || didNotExist.size() > 0) {
131             String msg = "";
132             String concat = "";
133             if (badGroupNames.size() > 0) {
134                 msg += context.getText("manager.error.badgroupnames") + ": " +
135                         StringUtil.convertCollectionToCommaDelimitedString(badGroupNames) + ".";
136                 concat = " ";
137             }
138 
139             if (didNotExist.size() > 0) {
140                 msg += concat;
141                 msg += context.getText("manager.error.groupsdidnotexist") + ": " +
142                         StringUtil.convertCollectionToCommaDelimitedString(didNotExist) + ".";
143             }
144 
145             if (success.size() > 0) {
146                 msg += concat;
147                 msg += context.getText("manager.error.removegroupspartialsuccess") + ": " +
148                         StringUtil.convertCollectionToCommaDelimitedString(success) + ".";
149             }
150             throw new RemoveException(msg);
151         }
152     }
